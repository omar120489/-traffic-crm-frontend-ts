# Create demo data seed script
cat > apps/core-api/prisma/seed-demo.ts << 'EOF'
import { PrismaClient } from '@prisma/client';

const prisma = new PrismaClient();

async function seedDemo() {
  // Create demo organization
  const org = await prisma.organization.upsert({
    where: { id: 'demo-org' },
    update: {},
    create: {
      id: 'demo-org',
      name: 'Demo Organization',
      slug: 'demo',
    },
  });

  // Create demo users
  const users = await Promise.all([
    prisma.user.upsert({
      where: { email: 'demo@traffic-crm.example.com' },
      update: {},
      create: {
        email: 'demo@traffic-crm.example.com',
        name: 'Demo User',
        orgId: org.id,
        roles: ['admin'],
      },
    }),
  ]);

  // Create demo companies
  const companies = await Promise.all([
    prisma.company.upsert({
      where: { id: 'demo-acme' },
      update: {},
      create: {
        id: 'demo-acme',
        name: 'Acme Corporation',
        industry: 'Technology',
        website: 'https://acme.example.com',
        orgId: org.id,
      },
    }),
  ]);

  // Create demo contacts
  await Promise.all([
    prisma.contact.upsert({
      where: { id: 'demo-contact-1' },
      update: {},
      create: {
        id: 'demo-contact-1',
        name: 'John Smith',
        email: 'john@acme.example.com',
        title: 'CEO',
        companyId: companies[0].id,
        orgId: org.id,
      },
    }),
  ]);

  // Create demo pipeline
  const pipeline = await prisma.pipeline.upsert({
    where: { id: 'demo-pipeline' },
    update: {},
    create: {
      id: 'demo-pipeline',
      name: 'Sales Pipeline',
      orgId: org.id,
    },
  });

  // Create demo stages
  const stages = await Promise.all([
    prisma.stage.upsert({
      where: { id: 'demo-stage-lead' },
      update: {},
      create: {
        id: 'demo-stage-lead',
        name: 'Lead',
        position: 0,
        pipelineId: pipeline.id,
      },
    }),
    prisma.stage.upsert({
      where: { id: 'demo-stage-qualified' },
      update: {},
      create: {
        id: 'demo-stage-qualified',
        name: 'Qualified',
        position: 1,
        pipelineId: pipeline.id,
      },
    }),
  ]);

  // Create demo deals
  await Promise.all([
    prisma.deal.upsert({
      where: { id: 'demo-deal-1' },
      update: {},
      create: {
        id: 'demo-deal-1',
        name: 'Enterprise License',
        amountCents: 50000_00,
        stageId: stages[0].id,
        companyId: companies[0].id,
        ownerId: users[0].id,
        orgId: org.id,
      },
    }),
  ]);

  console.log('âœ… Demo data seeded successfully');
}

seedDemo()
  .catch((e) => {
    console.error(e);
    process.exit(1);
  })
  .finally(async () => {
    await prisma.$disconnect();
  });
EOF
